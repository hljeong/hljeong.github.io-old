1472a - cards for friends

problem statement

given a \(w \times h\) sheet of paper, one can:

  • cut and get two sheets of size \(\frac{w}{2} \times h\) if \(w\) is even

  • cut and get two sheets of size \(w \times \frac{h}{2}\) if \(h\) is even

print "YES" if it is possible to cut the \(w \times h\) sheet of paper into \(\geq n\) pieces, "NO" otherwise.

solution

we can cut lengthwise as long as the width is even and vice versa, and every cut doubles the amount of pieces.

code

c++ 
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);

    int t;
    cin >> t;
    while (t --> 0) {
        int w, h, n, p = 0;
        cin >> w >> h >> n;

        while (w % 2 == 0) {
            ++p;
            w /= 2;
        }

        while (h % 2 == 0) {
            ++p;
            h /= 2;
        }

        // p is actually never > 31 given that w, h <= 1e4
        if (p > 31 or n <= (1 << p)) {
            cout << "YES" << endl;
        } else {
            cout << "NO" << endl;
        }
    }
}

    }

    cout << "NO" << endl;

1472c - long jumps

problem statement

solution

code

c++ (fragment) 
const int MAXN = 2e5 + 5;
int a[MAXN];
long long score[MAXN];

int n;
cin >> n;
long long ans = 0;
for (int i = 0; i < n; ++i) {
    cin >> a[i];
}

for (int i = n - 1; i >= 0; --i) {
    score[i] = a[i];

    if (i + a[i] < n) {
        score[i] += score[i + a[i]];
    }

    if (score[i] > ans) {
        ans = score[i];
    }
}

cout << ans << endl;

solution

const int MAXN = 2e5 + 5;
int a[MAXN];
long long score[2];

int n;
cin >> n;
for (int i = 0; i < n; ++i) {
    cin >> a[i];
}
sort(a, a + n, greater<int>());

score[0] = score[1] = 0;
for (int i = 0; i < n; ++i) {
    if (a[i] % 2 == i % 2) {
        score[i % 2] += a[i];
    }
}

if (score[0] == score[1]) {
    cout << "Tie" << endl;
} else if (score[0] > score[1]) {
    cout << "Alice" << endl;
} else {
    cout << "Bob" << endl;
}

1472e - correct placement

problem statement

solution

code

c++ (fragment) 
const int MAXN = 2e5 + 5, INF = 1e9 + 7;
int n, w[MAXN], h[MAXN], tmp[2 * MAXN], fw_val[MAXN], fw_ind[MAXN];

void upd(int i, int val, int ind) {
    ++i;
    while (i < n) {
        if (val < fw_val[i]) {
            fw_val[i] = val;
            fw_ind[i] = ind;
        }
        i += i & -i;
    }
}

void qry(int i, int &val, int &ind) {
    val = INF;
    ind = -1;
    while (i != 0) {
        if (fw_val[i] < val) {
            val = fw_val[i];
            ind = fw_ind[i];
        }
        i -= i & -i;
    }
}

for (int i = 0; i <= n; ++i) {
    fw_val[i] = INF;
    fw_ind[i] = -1;
}

cin >> n;
for (int i = 0; i < n; ++i) {
    cin >> w[i] >> h[i];
    tmp[i] = w[i];
    tmp[n + i] = h[i];
}

// compression: https://codeforces.com/blog/entry/84164
sort(tmp, tmp + 2 * n);
unordered_map<int, int> mp;
int id = 0;
mp[tmp[0]] = id++;
for (int i = 1; i < 2 * n; ++i) {
    if (tmp[i] != tmp[i - 1]) {
        mp[tmp[i]] = id++;
    }
}

for (int i = 0; i < n; ++i) {
    upd(mp[w[i]], h[i], i + 1);
    upd(mp[h[i]], w[i], i + 1);
}

for (int i = 0; i < n; ++i) {
    int val, ind;

    qry(mp[w[i]], val, ind);
    if (val < h[i]) {
        cout << ind << ' ';
        continue;
    }

    qry(mp[h[i]], val, ind);
    if (val < w[i]) {
        cout << ind << ' ';
        continue;
    }

    cout << -1 << ' ';
}

1472f - new year's puzzle

problem statement

solution

code

c++ (fragment) 

int n, m;
cin >> n >> m;
vector<pair<int, int>> block;
for (int i = 0; i < m; ++i) {
    int row, col;
    cin >> row >> col;
    block.push_back(make_pair(col - 1, row - 1));
}

sort(block.begin(), block.end());
int profile = 0, ind = 0, last = 0;
while (ind < m) {
    if (profile != 0) {
        profile = 1 + (profile - 1 + block[ind].second - last) % 2;
    }
    int blockage = 1 << block[ind].first;
    if (ind + 1 < m && block[ind].second == block[ind + 1].second) {
        blockage = 3;
        ++ind;
    }
    if (profile == blockage || profile != 0 && blockage == 3) {
        cout << "NO" << endl;
        return;
    }
    profile = 3 - (profile + blockage);
    last = block[ind][1] + 1;
    ++ind;
}

if (profile == 0) {
    cout << "YES" << endl;
} else {
    cout << "NO" << endl;
}

1472g - moving to the capital

problem statement

solution

code

c++ (fragment) 
int MAXN = 2e5 + 5;
int queue[MAXN], queue_head, queue_tail, rank[MAXN], ans[MAXN];
bool visited[MAXN];

int n, m;
cin >> n >> m;
for (int i = 0; i < n; ++i) {
    visited[i] = false;
}
vector<vector<int>> graph(n);
for (int i = 0; i < m; ++i) {
    int u, v;
    cin >> u >> v;
    graph[u - 1].push_back(v - 1);
}

queue_head = queue_tail = 0;
visited[0] = true;
rank[0] = 0;
queue[queue_tail++] = 0;
while (queue_head != queue_tail) {
    int node = queue[queue_head++];
    for (int neighbor : graph[node]) {
        if (!visited[neighbor]) {
            visited[neighbor] = true;
            rank[neighbor] = rank[node] + 1;
            queue[queue_tail++] = neighbor;
        }
    }
}

vector<pair<int, int>> order;
for (int i = 0; i < n; ++i) {
    ans[i] = rank[i];
    order.push_back(make_pair(rank[i], i));
}
sort(order.begin(), order.end());

for (pair<int, int> entry : order) {
    int node = entry.second;
    for (int neighbor : graph[node]) {
        if (rank[neighbor] <= rank[node]) {
            if (rank[neighbor] < ans[node]) {
                ans[node] = rank[neighbor];
            }
        } else {
            if (ans[neighbor] < ans[node]) {
                ans[node] = ans[neighbor];
            }
        }
    }
}

for (int i = 0; i < n; ++i) {
    cout << ans[i] << " \n"[i == n - 1];
}
