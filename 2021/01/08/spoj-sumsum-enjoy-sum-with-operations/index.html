spoj sumsum - enjoy sum with operations

problem statement

given an array \(a\) of length \(n\), perform \(m\) queries of 2 types:

  1. set \(a_i\) to \(x\)
  2. compute the sum of the bitwise OR (or AND or XOR) of all pairs of elements in \(\{a_i, a_{i + 1}, \dots, a_j \}\)

solution

let's look at the second type of queries. first, note that since they only concern bitwise operations, we can compute the contribution of each bit independently.

the query then simplifies to "given an array \(b\) of bits of length \(n\), find the number of pairs of elements that have a bitwise OR (or AND or XOR) of 1 in \(\{b_i, b_{i + 1}, \dots, b_j\}\)." we can easily find this value if we know the number of elements, \(k = j - i + 1\), the number of 0s, \(cnt0\), and the number of 1s, \(cnt1\), in \(\{b_i, b_{i + 1}, \dots, b_j\}\):

  • \(\text{number of pairs that have a bitwise OR of 1} = \begin{pmatrix} k \\ 2 \end{pmatrix} - \begin{pmatrix} cnt0 \\ 2 \end{pmatrix}\)
  • \(\text{number of pairs that have a bitwise AND of 1} = \begin{pmatrix} cnt1 \\ 2 \end{pmatrix}\)
  • \(\text{number of pairs that have a bitwise XOR of 1} = cnt0 \cdot cnt1\)

then, considering both types of queries, we need to efficiently perform point updates and compute \(cnt0\) and \(cnt1\) for given ranges, and we can do so with a segment tree or a fenwick tree.

code

c++ \(\mathcal O(m \log n \log(\max(a)))\) 
#include <bits/stdc++.h>
using namespace std;
#define ll long long

const int MAXN = 1e5 + 5, MAXB = 30;
int a[MAXN], fenwick[MAXB][MAXN];

void upd(int i, int x) {
    for (int bit = 0; bit < MAXB; ++bit) {
        int diff = (x >> bit & 1) - (a[i] >> bit & 1);
        for (int ind = i + 1; ind < MAXN; ind += ind & -ind) {
            fenwick[bit][ind] += diff;
        }
    }
    a[i] = x;
}

// returns counts for each bit in a[:i]
vector<int> qry(int i) {
    vector<int> ans(MAXB);
    for (int bit = 0; bit < MAXB; ++bit) {
        for (int ind = i; ind != 0; ind -= ind & -ind) {
            ans[bit] += fenwick[bit][ind];
        }
    }
    return ans;
}

// returns counts for each bit in a[i : j]
vector<int> qry(int i, int j) {
    vector<int> ans = qry(j), subtract = qry(i);
    for (int bit = 0; bit < MAXB; ++bit) {
        ans[bit] -= subtract[bit];
    }
    return ans;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);

    int n, m;
    cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        int x;
        cin >> x;
        upd(i, x);
    }

    while (m --> 0) {
        int t;
        cin >> t;
        if (t == 1) {
            int x, i;
            cin >> x >> i;
            --i;
            upd(i, x);
        } else {
            string op;
            int i, j;
            cin >> op >> i >> j;
            --i;

            vector<int> cnt = qry(i, j);
            ll ans = 0, k = j - i;
            for (int bit = 0; bit < MAXB; ++bit) {
                ll cnt1 = cnt[bit], cnt0 = k - cnt1;
                if (op == "OR") {
                    ans += (k * (k - 1) / 2 - cnt0 * (cnt0 - 1) / 2) << bit;
                } else if (op == "AND") {
                    ans += cnt1 * (cnt1 - 1) / 2 << bit;
                } else {
                    ans += cnt0 * cnt1 << bit;
                }
            }
            cout << ans << endl;
        }
    }
}
